CREATE TABLE BANK_ACCOUNTS
(
ACCOUNT_ID NUMBER ,
BANK_ACC_NO VARCHAR(30) UNIQUE,
RECON_TYPE VARCHAR(30),
CAT_CODE VARCHAR(20),
SUB_ACC_NO VARCHAR(30),
STATUS VARCHAR(10) DEFAULT 'ACTIVE',
PRIMARY KEY (ACCOUNT_ID)
);


CREATE TABLE ITEM (
ITEM_ID NUMBER PRIMARY KEY,
BANK_ACC_NO VARCHAR(30),
TXN_REFERENCE VARCHAR(50),
TXN_AMOUNT NUMBER(12,4),
DRORCR VARCHAR(1),
TXN_CCY VARCHAR(3),
STMT_DATE DATE,
SYSTEM_ENTRY_DATE TIMESTAMP,
VALUE_DATE DATE,
ITEM_TYPE VARCHAR(20),
TXN_STOCK_CODE VARCHAR(12),
STOCK_ID NUMBER,
QUANTITY NUMBER(12,4),
PRICE NUMBER(12,4),
CONSTRAINT FK_BANK_ACC_NO FOREIGN KEY (BANK_ACC_NO) REFERENCES BANK_ACCOUNTS(BANK_ACC_NO)
);


CREATE TABLE ASSET_REGISTER 
(
STOCK_ID NUMBER PRIMARY KEY,
STOCK_CODE VARCHAR(12) UNIQUE,
DESCRIPTION VARCHAR(100),
ASSET_TYPE VARCHAR(20)
);

CREATE TABLE CLOSING_BALANCES (
CB_ID NUMBER PRIMARY KEY,
BANK_ACC_NO VARCHAR(30),
CBAL_DATE DATE,
CBAL_AMOUNT NUMBER(12,4),
CURRENCY_CODE VARCHAR(3)
);
